  36. /**
  37.  * This class contains a wrapper around the Ghostscript-Application.
  38.  *
  39.  * This needs the Ghostscript application to be installed on the server. If the
  40.  * gs-executable is not available the class will not be able to execute anything
  41.  *
  42.  * A working example might look like the following code:
  43.  * <code>
  44.  *
  45.  * // First we describe the output-format
  46.  * $device = new Org_Heigl_Ghostscript_Device_Jpeg ();
  47.  *
  48.  * // Set the JPEG-Quality to 100
  49.  * $device -> setQuality ( 100 );
  50.  *
  51.  * // Next we Create the ghostscript-Wrapper
  52.  * $gs = new Org_Heigl_Ghostscript ();
  53.  *
  54.  * // Set the device
  55.  * $gs -> setDevice ( $device )
  56.  * // Set the input file
  57.  *     -> setInputFile ( 'path/to/my/ps/or/pdf/file' )
  58.  * // Set the output file that will be created in the same directory as the input
  59.  *     -> setOutputFile ( 'output' )
  60.  * // Set the resolution to 96 pixel per inch
  61.  *     -> setResolution ( 96 )
  62.  * // Set Text-antialiasing to the highest level
  63.  *     -> setTextAntiAliasing ( Org_Heigl_Ghostscript::ANTIALIASING_HIGH );
  64.  *
  65.  * // convert the input file to an image
  66.  * if ( true === $gs -> render () ) {
  67.  *     echo 'success';
  68.  * } else {
  69.  *     echo 'some error occured';
  70.  * }
  71.  * </code>
  72.  *
  73.  * Alternatively the example could read as follows
  74.  * <code>
  75.  *
  76.  * // Create the ghostscript-Wrapper
  77.  * $gs = new Org_Heigl_Ghostscript ();
  78.  *
  79.  * // Set the device
  80.  * $gs -> setDevice ( 'jpeg' )
  81.  * // Set the input file
  82.  *     -> setInputFile ( 'path/to/my/ps/or/pdf/file' )
  83.  * // Set the output file that will be created in the same directory as the input
  84.  *     -> setOutputFile ( 'output' )
  85.  * // Set the resolution to 96 pixel per inch
  86.  *     -> setResolution ( 96 )
  87.  * // Set Text-antialiasing to the highest level
  88.  *     -> setTextAntiAliasing ( Org_Heigl_Ghostscript::ANTIALIASING_HIGH );
  89.  *
  90.  * // Set the jpeg-quality to 100
  91.  * $gs -> getDevice () -> setQuality ( 100 );
  92.  *
  93.  * // convert the input file to an image
  94.  * if ( true === $gs -> render () ) {
  95.  *     echo 'success';
  96.  * } else {
  97.  *     echo 'some error occured';
  98.  * }
  99.  * </code>

  186.      * PortableDocumentformat (pdf) File.

  188.      * @param string|SplFileInfo$file The File to use as input.

  192.      */
  193.     public function setInputFile ( $file ) {
  194.         if ( ! $file instanceof SplFileInfo ) {
  195.             $file = new SplFileInfo ( (string) $file );
  196.         }
  197.         if ( extension_loaded ( 'fileinfo' ) ) {
  198.             $finfo = new finfo ();
  199.             $mime = $finfo -> file ( $file -> getPathName (), FILEINFO_MIME );
  200.             if ( ! in_array ( $mime, Org_Heigl_Ghostscript::$supportedMimeTypes ) ) {
  201.                 throw new InvalidArgumentException ( 'The provided file seems not to be of a supported MIME-Type' );
  202.             }
  203.         }
  204.         $this -> _infile = $file;
  205.         return $this;
  206.     }

  217.     /**
  218.      * Set the name of the output file(s)

  231.     public function setOutputFile ( $name = 'output' ) {
  232.         if ( 0 !== strpos ( $name, DIRECTORY_SEPARATOR ) ) {
  233.             $name = $this -> getBasePath () . DIRECTORY_SEPARATOR . $name;

  235.         $this -> _outfile = $name;
  236.     }

  247.     public function getOutputFile () {

  271.     /**
  272.      * Render the input file via Ghostscript
  273.      *
  274.      * @return bool 
  275.      */
  276.     public function render () {
  277.  
  278.         $renderString = $this -> getRenderString ();
  279.  
  280.         // We can't render anything without a render string
  281.         if ( '' == $renderString ) {
  282.             return false;
  283.         }
  284.  
  285.         exec ( $renderString , $returnArray, $returnValue );
  286.  
  287.         if ( 0 !== $returnValue ) {
  288.             return false;
  289.         }
  290.         return true;
  291.     }
  292.  
  293.     /**
  294.      * Get the command-line that can be executed via exec
  295.      *
  296.      * @return string 
  297.      */
  298.     public function getRenderString () {
  299.         if ( null === $this -> getInputFile () ) {
  300.             return '';
  301.         }
  302.         $string  = Org_Heigl_Ghostscript::getGsPath ();
  303.         $string .= ' -dSAFER -dQUIET -dNOPLATFONTS -dNOPAUSE -dBATCH';
  304.         $string .= ' -sOutputFile="' . $this -> getOutputFile () . '.' . $this -> getDevice () -> getFileEnding () . '"';
  305.         $string.=  $this -> getDevice () -> getParameterString ();
  306.         $string .= ' -r' . $this -> getResolution ();
  307.         if ( $this -> isTextAntiAliasingSet () ) {
  308.             $string .= ' -dTextAlphaBits=' . $this -> getTextAntiAliasing ();
  309.         }
  310.         if ( $this -> isGraphicsAntiAliasingSet () ) {
  311.             $string .= ' -dGraphicsAlphaBits=' . $this -> getGraphicsAntiAliasing ();
  312.         }
  313.         $string .= ' "' . $this -> getInputFile () . '"';
  314.         return $string;
  315.     }
